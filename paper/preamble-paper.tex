\usepackage[notheorems,noxcolor,noamsthm,noamssymb]{beamerarticle}

\usepackage[inline]{enumitem}
\setlist[enumerate]{label*={(\arabic*)}}

\newlist{properties@}{enumerate}{2}
\setlist[properties@]{label*=\arabic*.,ref=\arabic*}
\newlist{requirements@}{enumerate}{2}
\setlist[requirements@]{label*=\arabic*.,ref=\arabic*}
\setlist[requirements@,2]{ref=\arabic{requirements@i}.\arabic*}
\newlist{adversaries@}{enumerate}{2}
\setlist[adversaries@]{label*=A\arabic*.,ref=A\arabic*}
\setlist[adversaries@,2]{ref=A\arabic{adversaries@i}.\arabic*}

  \IfValueTF{#1}{%

  }{%
    \begin{properties@}%
  }
}{%
  \end{properties@}%
}

  \IfValueTF{#1}{%

  }{%
    \begin{requirements@}%
  }
}{%
  \end{requirements@}%
}
\DeclareDocumentEnvironment{adversaries}{o}{%
  \IfValueTF{#1}{%
    \setlist[adversaries@,2]{ref=#1\arabic{adversaries@i}.\arabic*}
    \begin{adversaries@}[resume=adversaries,label*=#1\arabic*.,ref=#1\arabic*]%
  }{%
    \begin{adversaries@}[resume=adversaries]%
  }
}{%
  \end{adversaries@}%
}
